§ Changing the Default Options
Change the default options by setting $.fn.immybox.defaults. Changing the default jQuery options will not affect the default options for instances created using new ImmyBox(element) 
$.fn.immybox.defaults = {
  maxResults: 5,
  showArrow: false,
  open_on_focus: false,
  defaultSelectedValue: 'LA',
  scroll_behavior: 'instant',
  no_results_text: 'no results',
  makeQueryResultsText(x, y) {
    return `${x} matching results (out of ${y})`;
  },
  filterFn(query) {
    let lower_query = query.toLowerCase();
    return choice => choice.alwaysShow || choice.text.toLowerCase().indexOf(lower_query) >= 0;
  },
  formatChoice(query) {
    if (query) {
      let reg = new RegExp(query.replace(/[#-.]|[[-^]|[?|{}]/g, '\\$&'), 'gi');
      return choice => choice.text.replace(reg, '<strong class="highlight">$&</strong>');
    } else
      return choice => choice.text;
  }
};

  ]
});

§ Simple Example

The  ImmyBox  constructor takes any  input[type="text"]  element as the first argument and an options  Object  as the second. 
// Tell ImmyBox to reposition when scrolling on #scrollable
ImmyBox.repositionWhenScrolling(document.getElementById('scrollable'));

// Pass an element to ImmyBox's constructor, along with an Object of options
new ImmyBox(document.getElementById('input-1'), {
  choices: [
    {text: 'Alabama', value: 'AL'},
    ...
    {text: 'Wyoming', value: 'WY'}
  ],
  defaultSelectedValue: 'LA'
});

§ choices
An array of choice objects used to populate the autocomplete list. Each object should have at least  text and value properties.
Default: [] 
let el = document.getElementById('input-2');
new ImmyBox(el, {
  choices: [
    {text: 'Foo', value: 'foo'},
    {text: 'Bar', value: 'bar'}
  ]
});

§ formatChoice
A function for formatting choices
Arguments:
 query - the string which the user entered into the text box
Return:
 (choice) => choice_string - function that takes in a choice and returns a string (which may contain HTML)
Default: choice.text with all parts that match query underlined 
let el = document.getElementById('input-6');
new ImmyBox(el, {
  choices: [
    {text: 'John Smith', value: 'jsmith', age: 31, employer: 'Smith Enterprises'},
    {text: 'Jane Doe', value: 'jdoe', age: 20, employer: 'National Security Agency'}
  ],
  formatChoice(query) {
    const reg = new RegExp("(" + query + ")", "i");
    return choice => (
      "<div class='mdl-grid mdl-grid--no-spacing'>" +
        "<div class='mdl-cell mdl-cell--6-col'>" +
          [
            choice.text.replace(reg, "<u>$1</u>"),
            choice.age
          ].join(", ") +
        "</div>" +
        "<div class='mdl-cell mdl-cell--6-col'>" +
          choice.employer +
        "</div>" +
      "</div>"
    );
  }
});
§ filterFn
A function for determining whether or not a choice matches the entered query
Arguments:
 query - the string which the user entered into the text box
Return:
 (choice) => is_match - function that takes in a choice and returns true if that choices matches query, or false if it does not
Default: case insensitive "contains" matching 
let el = document.getElementById('input-7');
new ImmyBox(el, {
  choices: [
    {text: 'John Smith', value: 'jsmith', age: 31, employer: 'Smith Enterprises'},
    {text: 'Jane Doe', value: 'jdoe', age: 20, employer: 'National Security Agency'}
  ],
  filterFn(query) {
    const lowercase_query = query.toLowerCase();
    return choice => (
      choice.text.toLowerCase().indexOf(lowercase_query) >= 0 ||
      String(choice.age).toLowerCase().indexOf(lowercase_query) >= 0 ||
      choice.employer.toLowerCase().indexOf(lowercase_query) >= 0
    );
  },
  formatChoice(query) {
    const reg = new RegExp("(" + query + ")", 'i');
    return choice => (
      "<div class='mdl-grid mdl-grid--no-spacing'>" +
        "<div class='mdl-cell mdl-cell--6-col'>" +
          [
            choice.text.replace(reg, "<u>$1</u>"),
            String(choice.age).replace(reg, "<u>$1</u>")
          ].join(", ") +
        "</div>" +
        "<div class='mdl-cell mdl-cell--6-col'>" +
          choice.employer.replace(reg, "<u>$1</u>") +
        "</div>" +
      "</div>"
    );
  }
});
§ defaultSelectedValue
The value that will be selected by default in the autocomplete list
Possible values:
 Object - the value of the choice; if not found in the choices array, it will act like undefined
 undefined - will tell immybox that nothing should be selected by default Default: undefined 
let el = document.getElementById('input-8');
new ImmyBox(el, {
  choices: [
    {text: 'Foo', value: 'foo'},
    {text: 'Bar', value: 'bar'}
  ],
  defaultSelectedValue: 'bar'
});
§ scrollBehavior
The behavior to use when an immybox is opened and the selected result scrolls into view
Possible values:
 'smooth' smoothly scroll the selected choice into view
 'instant' - open the ImmyBox directly to the selected choice
Default: 'smooth'

§ no_results_text
The text that is displayed when no results match the current query

§ makeQueryResultsText
A function for producing the 'showing x of y' text below the query result area. Accepts numVisibleChoices and numChoicesMatchingFilter as arguments and must return a string.
Possible values:
 Function - the function to use
 undefined|null|false|0 - do not display any text
Default: (x, y) => `showing ${x} of ${y}` 
let el = document.getElementById('input-11');
new ImmyBox(el, {
  choices: [
    {text: 'Alabama', value: 'AL'},
    ...
    {text: 'Wyoming', value: 'WY'}
  ],
  maxResults: 10,
  makeQueryResultsText(numVisibleChoices, numChoicesMatchingFilter) {
    if (numChoicesMatchingFilter === numVisibleChoices) {
      return `Showing all ${numChoicesMatchingFilter} matching states`;
    }
    return `${numChoicesMatchingFilter} states match; displaying ${numVisibleChoices} of them`;
  }
});
§ Changing the Default Options
Change the default options by setting ImmyBox.defaults.
Changing the default ImmyBox options will also change the default options for instances created using the jQuery plugin, but will not override $.fn.immybox.defaults
Use ImmyBox.resetDefaults() to undo changes made to the ImmyBox default options 
ImmyBox.defaults = {
  maxResults: 5,
  showArrow: false,
  open_on_focus: false,
  defaultSelectedValue: 'LA',
  scroll_behavior: 'instant',
  no_results_text: 'no results',
  makeQueryResultsText(x, y) {
    return `${x} matching results (out of ${y})`;
  },
  filterFn(query) {
    let lower_query = query.toLowerCase();
    return choice => choice.alwaysShow || choice.text.toLowerCase().indexOf(lower_query) >= 0;
  },
  formatChoice(query) {
    if (query) {
      let reg = new RegExp(query.replace(/[#-.]|[[-^]|[?|{}]/g, '\\$&'), 'gi');
      return choice => choice.text.replace(reg, '<strong class="highlight">$&</strong>');
    } else
      return choice => choice.text;
  }
};

let el = document.getElementById('input-12');
new ImmyBox(el, {
